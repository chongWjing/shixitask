## 10.1 Axios封装

> "为什么要封装Axios？因为每个页面都单独写axios调用太乱了！封装之后，我们只需要在请求拦截器里统一加token，在响应拦截器里统一处理错误，不用每个地方都重复写。"

### 步骤1：创建请求封装文件

在 `src/api/` 目录下创建 `request.js` 文件：

```javascript
// src/api/request.js
// Axios请求封装 —— 统一管理请求配置、拦截器

import axios from 'axios'
import { ElMessage } from 'element-plus'
import router from '../router'

// 创建axios实例
const service = axios.create({
  baseURL: '',  // 为空，走vite.config.js中配置的proxy代理
  timeout: 10000, // 请求超时时间10秒
  headers: {
    'Content-Type': 'application/json'
  }
})

// 请求拦截器
service.interceptors.request.use(
  config => {
    // 从localStorage获取token
    const token = localStorage.getItem('token')
    if (token) {
      // 如果token存在，自动在Header中添加Authorization字段（无Bearer前缀）
      config.headers['Authorization'] = token
    }
    return config
  },
  error => {
    // 请求错误处理
    console.error('请求错误：', error)
    return Promise.reject(error)
  }
)

// 响应拦截器
service.interceptors.response.use(
  response => {
    // 对响应数据进行处理
    const res = response.data
    // 如果后端返回的code不是200，说明有业务错误
    if (res.code !== 200) {
      ElMessage.error(res.message || '请求失败')
      // 401表示未登录或token过期
      if (res.code === 401) {
        // 清除本地存储的登录信息
        localStorage.removeItem('token')
        localStorage.removeItem('username')
        // 跳转到登录页
        router.push('/login')
      }
      return Promise.reject(new Error(res.message || '请求失败'))
    }
    // 正常返回数据
    return res
  },
  error => {
    // HTTP状态码错误处理
    console.error('响应错误：', error)
    if (error.response) {
      switch (error.response.status) {
        case 401:
          ElMessage.error('登录已过期，请重新登录')
          localStorage.removeItem('token')
          localStorage.removeItem('username')
          router.push('/login')
          break
        case 403:
          ElMessage.error('没有权限访问')
          break
        case 404:
          ElMessage.error('请求的资源不存在')
          break
        case 500:
          ElMessage.error('服务器内部错误')
          break
        default:
          ElMessage.error('网络错误：' + error.response.status)
      }
    } else {
      ElMessage.error('网络连接失败，请检查网络')
    }
    return Promise.reject(error)
  }
)

export default service
```

**预期效果：** 之后所有API请求都通过 `service` 发出，自动携带token，自动处理错误。

## 10.2 API模块编写

> "我们把接口按功能模块分类，放到不同的文件里，这样代码结构清晰，维护方便。就像图书馆把书分到不同书架一样！"

### 步骤2：创建用户API模块

创建文件 `src/api/user.js`：

```javascript
// src/api/user.js
// 用户相关接口

import request from './request'

// 用户登录
export function login(data) {
  return request({
    url: '/api/user/login',
    method: 'post',
    data: data
  })
}

// 用户注册
export function register(data) {
  return request({
    url: '/api/user/register',
    method: 'post',
    data: data
  })
}

// 获取用户信息
export function getUserInfo() {
  return request({
    url: '/api/user/info',
    method: 'get'
  })
}

// 退出登录
export function logout() {
  return request({
    url: '/api/user/logout',
    method: 'post'
  })
}
```

**预期效果：** 调用 `login({username: 'xxx', password: 'xxx'})` 即可发起登录请求。

### 步骤3：创建商品API模块

创建文件 `src/api/product.js`：

```javascript
// src/api/product.js
// 商品相关接口

import request from './request'

// 获取商品列表（支持分页参数）
// 对接后端接口：GET /api/product/list?page=1&size=10
export function getProductList(params) {
  return request({
    url: '/api/product/list',
    method: 'get',
    params: params
  })
}

// 获取商品详情
// 对接后端接口：GET /api/product/{id}
export function getProductDetail(id) {
  return request({
    url: '/api/product/' + id,
    method: 'get'
  })
}

// 搜索商品
// 对接后端接口：GET /api/product/search?keyword=xxx
export function searchProduct(keyword) {
  return request({
    url: '/api/product/search',
    method: 'get',
    params: { keyword: keyword }
  })
}
```

**预期效果：**
- 调用 `getProductList({page: 1, size: 10})` 可获取第一页商品列表
- 调用 `getProductDetail(1)` 可获取ID为1的商品详情，请求路径为 `/api/product/1`
- 调用 `searchProduct('红玫瑰')` 可搜索包含"红玫瑰"关键字的商品

import { ref } from 'vue'
import { useRouter } from 'vue-router'

// 当前标签页

  username: '',
  password: ''

const loginLoading = ref(false)

  username: [{ required: true, message: '请输入用户名', trigger: 'blur' }],
  password: [{ required: true, message: '请输入密码', trigger: 'blur' }]
}

// 注册表单
const registerFormRef = ref(null)
const registerForm = ref({
  username: '',
  password: '',
  confirmPassword: ''
})
const registerLoading = ref(false)

// 确认密码校验
const validateConfirmPassword = (rule, value, callback) => {
  if (value !== registerForm.value.password) {
    callback(new Error('两次密码不一致'))
  } else {
    callback()
  }
}

const registerRules = {
  username: [{ required: true, message: '请输入用户名', trigger: 'blur' }],
  password: [{ required: true, message: '请输入密码', trigger: 'blur' }],
  confirmPassword: [
    { required: true, message: '请确认密码', trigger: 'blur' },
    { validator: validateConfirmPassword, trigger: 'blur' }
  ]

const onLogin = async () => {
  try {
    await loginFormRef.value.validate()
  } catch {
    return
  }

  loginLoading.value = true
  try {
    const res = await login({
      username: loginForm.value.username,
      password: loginForm.value.password

    localStorage.setItem('token', res.data.token)
    localStorage.setItem('username', res.data.username)
    ElMessage.success('登录成功')
    // 跳转到首页
    router.push('/')
  } catch (error) {
    // 错误已在拦截器中处理，这里可以不做额外处理
    console.error('登录失败：', error)
  } finally {
    loginLoading.value = false
  }
}

// 注册
const onRegister = async () => {
  try {
    await registerFormRef.value.validate()
  } catch {
    return
  }

  registerLoading.value = true
  try {
    const res = await register({
      username: registerForm.value.username,
      password: registerForm.value.password
    })
    ElMessage.success('注册成功，请登录')
    // 切换到登录标签页
    activeTab.value = 'login'
    // 自动填入用户名
    loginForm.value.username = registerForm.value.username
    // 清空注册表单
    registerForm.value = {
      username: '',
      password: '',
      confirmPassword: ''
    }
  } catch (error) {
    console.error('注册失败：', error)
  } finally {
    registerLoading.value = false
  }
}
</script>

<script setup>
import { ref, onMounted } from 'vue'
import { useRouter } from 'vue-router'
import { getProductList, searchProduct } from '../api/product'

// 商品分类

const activeCategory = ref('全部')

// 商品列表数据

// 加载状态
const loading = ref(false)
const loadingMore = ref(false)
// 当前页码
const currentPage = ref(1)
// 每页条数
const pageSize = 12
// 是否还有更多
const hasMore = ref(true)
// 搜索关键词
const searchKeyword = ref('')

// 加载商品列表
const loadProducts = async (reset = false) => {
  if (reset) {
    currentPage.value = 1
    productList.value = []
    hasMore.value = true
  }

  const isLoadingMore = !reset
  if (isLoadingMore) {
    loadingMore.value = true
  } else {
    loading.value = true
  }

  try {
    const params = {
      page: currentPage.value,
      size: pageSize
    }
    if (activeCategory.value !== '全部') {
      params.category = activeCategory.value
    }

    const res = await getProductList(params)
    const list = res.data.list || res.data.records || res.data || []
    if (list.length > 0) {
      if (reset) {
        productList.value = list
      } else {
        productList.value = [...productList.value, ...list]
      }
      currentPage.value++
    }
    if (list.length < pageSize) {
      hasMore.value = false
    }
  } catch (error) {
    console.error('加载商品列表失败：', error)
    hasMore.value = false
  } finally {
    loading.value = false
    loadingMore.value = false
  }
}

// 加载更多
const loadMore = () => {
  loadProducts(false)
}

// 分类点击
const onCategoryClick = (cat) => {
  activeCategory.value = cat
  loadProducts(true)
}

// 搜索
const onSearch = async () => {
  const keyword = searchKeyword.value.trim()
  if (!keyword) {
    loadProducts(true)
    return
  }
  loading.value = true
  try {
    const res = await searchProduct(keyword)
    productList.value = res.data.list || res.data.records || res.data || []
    hasMore.value = false
  } catch (error) {
    console.error('搜索商品失败：', error)
  } finally {
    loading.value = false
  }
}

// 跳转到商品详情
const goDetail = (id) => {
  router.push('/product/' + id)
}

// 页面加载
onMounted(() => {
  loadProducts(true)
})
</script>

> "前端路由需要区分商城页面和后台管理页面。商城页面走 `/shop/*` 路径使用 ShopLayout 包裹，后台管理走 `/admin/*` 路径使用 AdminLayout 包裹。"

### 步骤10：路由配置参考

在 `frontend/src/router/index.js` 中，路由结构如下：

```javascript
import { createRouter, createWebHistory } from 'vue-router'

// 商城页面
import ShopLayout from '../views/ShopLayout.vue'
import Home from '../views/Home.vue'
import Login from '../views/Login.vue'
import ProductDetail from '../views/ProductDetail.vue'
import Order from '../views/Order.vue'
import MyOrders from '../views/MyOrders.vue'

// 后台管理页面
import AdminLayout from '../views/admin/Layout.vue'
import AdminLogin from '../views/admin/Login.vue'
import Dashboard from '../views/admin/Dashboard.vue'
import ProductManage from '../views/admin/ProductManage.vue'
import OrderManage from '../views/admin/OrderManage.vue'
import AiQuery from '../views/admin/AiQuery.vue'

const routes = [
  // ========== 商城页面（/shop/* 使用 ShopLayout 包裹） ==========
  {
    path: '/shop',
    component: ShopLayout,
    children: [
      { path: '', redirect: '/shop/home' },
      { path: 'home', component: Home },
      { path: 'product/:id', component: ProductDetail },
      { path: 'order', component: Order },
      { path: 'myorders', component: MyOrders }
    ]
  },
  // 登录页
  { path: '/login', component: Login },
  // 首页重定向
  { path: '/', redirect: '/shop/home' },

  // ========== 后台管理页面（/admin/* 使用 AdminLayout 包裹） ==========
  { path: '/admin/login', component: AdminLogin },
  {
    path: '/admin',
    component: AdminLayout,
    meta: { requiresAdmin: true },
    children: [
      { path: '', redirect: '/admin/dashboard' },
      { path: 'dashboard', component: Dashboard, meta: { requiresAdmin: true } },
      { path: 'product', component: ProductManage, meta: { requiresAdmin: true } },
      { path: 'order', component: OrderManage, meta: { requiresAdmin: true } },
      { path: 'ai', component: AiQuery, meta: { requiresAdmin: true } }
    ]
  }
]

const router = createRouter({
  history: createWebHistory(),
  routes
})

// 路由守卫：管理后台需要管理员登录
router.beforeEach((to, from, next) => {
  const token = localStorage.getItem('token')
  const role = localStorage.getItem('role')

  // 管理后台需要管理员登录
  if (to.meta.requiresAdmin && (!token || role !== '1')) {
    next('/admin/login')
    return
  }

  next()
})

export default router
```

> "注意：商城页面使用 ShopLayout.vue 包裹，后台管理页面使用 AdminLayout（即 admin/Layout.vue）包裹。两种布局各自独立，互不影响。"

### 步骤14：常见调试技巧

> "联调过程中肯定会遇到问题，这里教大家几个排查方法。"

**1. 查看浏览器控制台Network面板**
- 打开浏览器开发者工具（F12），切换到Network标签
- 每次操作后查看请求是否发出、参数是否正确、响应是否正常

**2. 查看后端控制台日志**
- 观察后端控制台是否有异常堆栈信息
- 如果有SQL错误，检查Mapper XML中的SQL语句

**3. 检查token是否传递**
- 在Network面板中查看请求Header，确认Authorization字段存在
- 如果不存在，检查localStorage中是否有token

**4. 跨域问题排查**
- 确认 `vite.config.js` 中proxy配置正确
- 确认请求URL以 `/api` 开头（才能被proxy转发）

## 常见问题

| 问题 | 原因 | 解决方法 |
|------|------|----------|
| 请求返回401 | token无效或过期 | 重新登录获取新token |
| 请求返回404 | 接口URL不匹配 | 检查前端API路径和后端RequestMapping是否一致 |
| 请求返回500 | 后端代码异常 | 查看后端控制台错误日志，修复代码 |
| 商品列表为空 | 后端数据库无数据 | 往product表插入测试数据（红玫瑰、康乃馨等鲜花商品） |
| 登录成功但跳转后仍提示未登录 | token未正确存储 | 检查localStorage是否成功写入token |
| 商品列表数据重复 | 没有重置页码和数据列表 | 确保loadProducts(reset=true)中先清空productList并重置currentPage |
| 商品详情404 | 接口路径写错 | 确认使用 `/api/product/{id}` 而非 `/api/product/detail/{id}` |
| 下单失败 | 字段名不匹配 | 确认使用 receiverName/receiverPhone/receiverAddress |
| Network中请求URL不对 | 没走proxy或路径写错 | 确认baseURL为空，URL以/api开头 |
| 图片加载失败 | 图片URL无效 | 检查数据库中image字段是否为有效URL |
| ElementPlus组件样式异常 | 未正确引入ElementPlus | 确认main.js中引入了ElementPlus的CSS |

本章我们完成了以下工作：
